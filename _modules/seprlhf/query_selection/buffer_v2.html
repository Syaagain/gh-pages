class Buffer():
    def __init__(self, obs_shape, action_shape, max_size, device, n_envs):
        """
        Initializes the buffer with parameters for storing transitions.

        Parameters
        ----------
        obs_shape : tuple
            Shape of the observation space.
        action_shape : tuple
            Shape of the action space.
        max_size : int
            Maximum number of transitions the buffer can store.
        device : str or torch.device
            Device for PyTorch tensors (e.g., 'cpu' or 'cuda').
        n_envs : int
            Number of parallel environments.
        """

        self.max_size = max_size
        self.device = device
        self.obs_shape = obs_shape
        self.action_shape = action_shape
        self.n_envs = n_envs

        self.obs_buffer, self.next_obs_buffer, self.action_buffer, self.reward_buffer = None, None, None, None
        self.return_buffer, self.done_buffer, self.value_buffer, self.logprob_buffer = None, None, None, None

        self.idx = 0
        self.full = False
        self.generator_ready = False
        self.reset()

    def reset(self):
        """
        Resets the buffer by initializing the storage arrays.
        """

        self.obs_buffer = np.zeros(
            (self.max_size, self.n_envs) + self.obs_shape, dtype=np.float32)
        self.next_obs_buffer = np.zeros(
            (self.max_size, self.n_envs) + self.obs_shape, dtype=np.float32)
        self.action_buffer = np.zeros(
            (self.max_size, self.n_envs) + self.action_shape, dtype=np.float32)
        self.reward_buffer = np.zeros(
            (self.max_size, self.n_envs), dtype=np.float32)
        self.return_buffer = np.zeros(
            (self.max_size, self.n_envs), dtype=np.float32)
        self.done_buffer = np.zeros(
            (self.max_size, self.n_envs), dtype=np.float32)
        self.value_buffer = np.zeros(
            (self.max_size, self.n_envs), dtype=np.float32)
        self.logprob_buffer = np.zeros(
            (self.max_size, self.n_envs), dtype=np.float32)
        self.idx = 0
        self.full = False
        self.generator_ready = False



    def __len__(self):
        """
        Returns the current size of the buffer.

        Returns
        -------
        int: Maximum size if the buffer is full, otherwise the current index.
        """

        return self.max_size if self.full else self.idx

    def reshape(self, arr: np.ndarray):
        """
        Reshapes a multi-dimensional array for easier sampling.

        Parameters
        ----------
        arr : np.ndarray
            Input array to be reshaped.

        Returns
        -------
        np.ndarray
            Reshaped array.
        """

        shape = arr.shape
        if len(shape) < 3:
            shape = shape + (1,)
        return arr.swapaxes(0, 1).reshape(shape[0] * shape[1], *shape[2:])

    def store(self, obs, action, reward, next_obs, done, value, logprob):
        """
        Stores a new transition in the buffer.

        Parameters
        ----------
        obs : np.ndarray
            Current observation.
        action : np.ndarray
            Action taken by the agent.
        reward : np.ndarray
            Reward received after taking the action.
        next_obs : np.ndarray
            Next observation after taking the action.
        done : np.ndarray
            Flag indicating if the episode has terminated.
        value : torch.Tensor
            Estimated value of the current state.
        logprob : torch.Tensor
            Log probability of the action.
        """

        if len(logprob.shape) == 0:
            logprob = logprob.reshape(-1, 1)

        self.obs_buffer[self.idx] = obs.clone().cpu().numpy()
        self.action_buffer[self.idx] = action.clone().cpu().numpy()
        self.reward_buffer[self.idx] = reward.clone().cpu().numpy()
        self.next_obs_buffer[self.idx] = next_obs.clone().cpu().numpy()
        self.done_buffer[self.idx] = done.clone().cpu().numpy()
        self.value_buffer[self.idx] = value.clone().cpu().numpy().flatten()
        self.logprob_buffer[self.idx] = logprob.clone().cpu().numpy()

        self.idx += 1
        if self.idx == self.max_size:
            self.full = True

    def sample(self, num_segments: int):
        """
        Samples batch of transitions from the buffer.

        Parameters
        ----------
        num_segments : int
            Number of transitions in each sampled segment.

        Yields
        ------
        Tuple[torch.Tensor, torch.Tensor, torch.Tensor, torch.Tensor, torch.Tensor]
            Sequential batches of transitions containing observations, actions, rewards, log probabilities, and returns.
        """
        assert self.full, "Buffer is not full yet."

        idxs = np.arange(self.max_size * self.n_envs)

        if not self.generator_ready:
            for tensor in ["obs_buffer", "action_buffer", "reward_buffer", "logprob_buffer", "return_buffer"]:
                self.__dict__[tensor] = self.reshape(self.__dict__[tensor])
            self.generator_ready = True

        start_idx = random.randint(0, self.max_size - num_segments)
        while start_idx + num_segments <= self.max_size * self.n_envs:
            segment_idxs = idxs[start_idx: start_idx + num_segments]
            yield self._get_samples(segment_idxs)
            start_idx += num_segments



    def _get_samples(self, segment_inds):
        """
        Retrieves samples from the buffer based on given indices.

        Parameters
        ----------
        segmment_inds : np.ndarray
            Indices of the transitions to sample.

        Returns
        -------
        Tuple[torch.Tensor, torch.Tensor, torch.Tensor, torch.Tensor, torch.Tensor]
            Tensors containing sampled transitions.
        """

        data = (
            torch.tensor(self.obs_buffer[segment_inds], device=self.device),
            torch.tensor(self.action_buffer[segment_inds], device=self.device),
            torch.as_tensor(
                self.reward_buffer[segment_inds], device=self.device),
            torch.as_tensor(
                self.logprob_buffer[segment_inds].flatten(), device=self.device),
            torch.tensor(
                self.return_buffer[segment_inds].flatten(), device=self.device),
        )
        return tuple(data)

    def compare_and_label(self, seg_1: tuple, seg_2: tuple):
        """
        Compares two batches of transitions based on their rewards and assigns labels.

        Parameters
        ----------
        seg_1 : tuple
            First batch of transitions.
        seg_2 : tuple
            Second batch of transitions.

        Returns
        -------
        Tuple[torch.Tensor, torch.Tensor]
            Labels indicating which batch has higher or lower rewards.
        """
        label_1 = (torch.sum(seg_1[2]) > torch.sum(seg_2[2])).int()
        label_2 = (torch.sum(seg_1[2]) <= torch.sum(seg_2[2])).int()
        return label_1, label_2

    def compare_and_label_hf(self, delete_videos: bool, video_loading_dir=None):
        """
        Compares video segments based on user preference and assigns labels.

        Parameters
        ----------
        delete_videos : bool
            Indicates whether to delete the video files and their parent directory after processing.
        video_loading_dir : str, optional
            Path to the directory containing video files to be processed. If None, no videos are processed.

        Returns
        -------
        Tuple[torch.Tensor, torch.Tensor]
            Labels indicating user preference for the video segments:
        """

        if video_loading_dir is not None:
            user_choice = wait_for_user_preference(video_loading_dir)
            if user_choice == 1:
                label_1 = torch.tensor(1, dtype=torch.int, device=self.device)
                label_2 = torch.tensor(0, dtype=torch.int, device=self.device)
            elif user_choice == 2:
                label_1 = torch.tensor(0, dtype=torch.int, device=self.device)
                label_2 = torch.tensor(1, dtype=torch.int, device=self.device)
            else:
                label_1 = torch.tensor(0, dtype=torch.int, device=self.device)
                label_2 = torch.tensor(0, dtype=torch.int, device=self.device)

        if delete_videos:
            video_path = Path(video_loading_dir).glob("video_seg*.mp4")
            for video in video_path:
                os.remove(video)
            try:
                shutil.rmtree(Path(video_loading_dir).parent)
            except OSError:
                pass
        return label_1, label_2

    def get_obs_action(self, seg):
        """
        Extracts and reshapes observations and actions from a segment.

        Parameters
        ----------
        seg : tuple
            A tuple containing the sampled observations, actions, ...

        Returns
        -------
        Tuple[torch.Tensor, torch.Tensor]
            A tuple containing:
                - obs_tensor: A reshaped tensor of observations with shape (trajectory_length, obs_dim).
                - actions_tensor: A reshaped tensor of actions with shape (trajectory_length, act_dim).
        """

        obs_action = []

        trajectory_length = seg[0].shape[0]
        obs_dim = seg[0].shape[2]
        act_dim = seg[1].shape[2]

        obs_reshaped = seg[0].reshape(-1, obs_dim)
        actions_reshaped = seg[1].reshape(-1, act_dim)
        trajectory_length = len(obs_reshaped)

        obs_tensor = torch.stack([obs_reshaped[i]
                                 for i in range(trajectory_length)])
        actions_tensor = torch.stack(
            [actions_reshaped[i] for i in range(trajectory_length)])

        return obs_tensor, actions_tensor

    def reward_based_inds(self, num_preference_pairs: int, num_segments: int):
        """
        Selects segments based on entropy-like criteria and ranks them by cumulative rewards.
        This function samples random segments of data indices, calculates their cumulative rewards,
        and sorts them in ascending order based on the cumulative rewards.

        Parameters
        ----------
        num_preference_pairs : int
            The number of preference pairs to generate. Each pair corresponds to two segments.
        num_segments : int
            The number of segments to sample for each preference.

        Returns
        -------
        Tuple[list, list]
            A tuple containing:
            - idx_list: A list of indices sorted by cumulative reward.
            - segments: A list of sampled segment indices.
        """

        cumulative_rewards = []
        segments = []
        for i in range(0, num_preference_pairs*2):
            idxs = np.arange(self.max_size * self.n_envs)
            segment_inds = random.choices(idxs, k=num_segments)
            segments.append(segment_inds)
            rewards = self.reward_buffer[segment_inds]
            cumulative_rewards.append(np.sum(rewards))
        
        idx_list = [inds[0] for inds in sorted(enumerate(cumulative_rewards), key=lambda i: i[1])] 
        return (idx_list, segments)
