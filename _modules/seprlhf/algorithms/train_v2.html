@dataclass
class Args:
    exp_name: str = os.path.basename(__file__)[: -len(".py")]
    """the name of this experiment"""
    seed: int = 1
    """seed of the experiment"""
    torch_deterministic: bool = True
    """if toggled, `torch.backends.cudnn.deterministic=False`"""
    cuda: bool = True
    """if toggled, cuda will be enabled by default"""
    track: bool = False
    """if toggled, this experiment will be tracked with Weights and Biases"""
    wandb_project_name: str = "cleanRL"
    """the wandb's project name"""
    wandb_entity: str = None
    """the entity (team) of wandb's project"""
    capture_video: bool = False
    """whether to capture videos of the agent performances (check out `videos` folder)"""
    save_model: bool = False
    """whether to save model into the `runs/{run_name}` folder"""
    upload_model: bool = False
    """whether to upload the saved model to huggingface"""
    hf_entity: str = ""
    """the user or org name of the model repository from the Hugging Face Hub"""
    human_feedback: bool = False
    """whether to use human or synthetic feedback"""
    delete_videos: bool = True
    """Automatically deletes captured videos after feedback was collected"""
    matplotlib: bool = True
    """Whether to plot the reward with matplotlib"""
    pretrain: bool = True
    """Whether to pretrain the reward model"""

    # Algorithm specific arguments
    env_id: str = "HalfCheetah-v4"
    """the id of the environment"""
    total_timesteps: int = 10000000
    """total timesteps of the experiments"""
    learning_rate: float = 3e-4
    """the learning rate of the optimizer"""
    rm_learning_rate: float = 3e-4  # davor -4
    """the learning rate of the reward model optimizer"""
    num_envs: int = 1
    """the number of parallel game environments"""
    num_steps: int = 2048
    """the number of steps to run in each environment per policy rollout"""
    max_episode_steps: int = 1000
    """the maximum number of steps per episode"""
    anneal_lr: bool = True
    """Toggle learning rate annealing for policy and value networks"""
    anneal_rm_lr: bool = True
    """Toggle learning rate annealing for the reward model networks"""
    init_agent: bool = False
    """Specifiy if the agents neural network layers should be initialized the same as in ppo."""
    init_reward_model: bool = False
    """Specifiy if the reward models neural networks layers should be initialized as the agent network in ppo."""
    gamma: float = 0.99
    """the discount factor gamma"""
    gae_lambda: float = 0.95
    """the lambda for the general advantage estimation"""
    num_minibatches: int = 32
    """the number of mini-batches"""
    update_epochs: int = 10

    norm_adv: bool = True
    """Toggles advantages normalization"""

    # TRY NOT TO MODIFY: seeding
    random.seed(args.seed)

    torch.manual_seed(args.seed)
    torch.backends.cudnn.deterministic = args.torch_deterministic

    device = torch.device(

    # Environment setup for gym
    envs = make_gymnasium_env(
        env_id=args.env_id,
        num_envs=args.num_envs,
        capture_video=args.capture_video,
        run_name=run_name,
        gamma=args.gamma,
    )

    assert isinstance(envs.single_action_space,
                      gym.spaces.Box), "only continuous action space is supported"

    agent = PPOAgent(envs, init_layer=args.init_agent).to(device)
    optimizer = optim.Adam(agent.parameters(), lr=args.learning_rate, eps=1e-5)

    reward_models, optimizers = create_reward_models(
        args.ensamble_size, envs, device, args)

    print(f"Agent Parameters: {count_parameters(agent)}")
    for rm in reward_models:
        print(f"Reward-Model Parameters: {count_parameters(rm)}")

    # ALGO Logic: Storage setup
    obs = torch.zeros((args.num_steps, args.num_envs) +
                      envs.single_observation_space.shape).to(device)
    actions = torch.zeros((args.num_steps, args.num_envs) +
                          envs.single_action_space.shape).to(device)
    logprobs = torch.zeros((args.num_steps, args.num_envs)).to(device)
    gt_rewards = torch.zeros((args.num_steps, args.num_envs)).to(device)
    rm_rewards = torch.zeros((args.num_steps, args.num_envs)).to(device)
    dones = torch.zeros((args.num_steps, args.num_envs)).to(device)
    values = torch.zeros((args.num_steps, args.num_envs)).to(device)

    # Reward model loss initialisation
    rm_loss = PreferenceLoss(
        reward_models=reward_models,
        device=device,
        normalize=True,
        batch_size=512,
    ).to(device)

    # Buffer creation and initialisation for the preference pairs
    obs_shape = envs.observation_space.shape
    action_shape = envs.action_space.shape
    max_size = 2048
    buffer_1 = Buffer(obs_shape, action_shape, max_size,
                      device, n_envs=args.num_envs)

    if args.human_feedback:
        frontend_thread = threading.Thread(
            target=lambda: uvicorn.run(
                app, host="127.0.0.1", port=8000, log_level="info"),
            daemon=True
        )
        frontend_thread.start()

    # TRY NOT TO MODIFY: start the game
    global_step = 0
    start_time = time.time()
    next_obs, _ = envs.reset(seed=args.seed)
    next_obs = torch.Tensor(next_obs).to(device)
    next_done = torch.zeros(args.num_envs).to(device)

    loss_fn = PPOLoss(
        clip_coef=args.clip_coef,
        clip_vloss=args.clip_vloss,
        ent_coef=args.ent_coef,
        vf_coef=args.vf_coef,
        norm_adv=args.norm_adv
    ).to(device)

    # Dynamically calculate the pretraining prefernce pairs amount, as well as the sampling intervalls for the training loop
    num_pre_train_preference_pairs = int(
        (args.total_num_preference_pairs * args.pretrain_sample_ratio) / args.num_preference_pairs)

    num_train_preference_pairs = args.total_num_preference_pairs - \
        (num_pre_train_preference_pairs * args.num_preference_pairs)

    num_validation_pairs = int(
        args.validation_ratio * num_train_preference_pairs)

    preference_pairs = []
    validation_pairs = []

    # Start of the pretraining loop, if pretraining is activated
    if args.pretrain:
        for step_pt in tqdm(range(1, num_pre_train_preference_pairs + 1), desc=f"Pre-Training over {num_pre_train_preference_pairs} steps..."):
            for step in range(0, args.num_steps):
                obs[step] = next_obs
                dones[step] = next_done

                with torch.no_grad():
                    action, logprob, _, value = agent.get_action_and_value(
                        next_obs)
                    values[step] = value.flatten()
                actions[step] = action
                logprobs[step] = logprob

                next_obs, gt_reward, terminations, truncations, infos = envs.step(
                    action.cpu().numpy())

                next_done = np.logical_or(terminations, truncations)
                gt_rewards[step] = torch.tensor(gt_reward).to(device).view(-1)
                next_obs, next_done = torch.Tensor(next_obs).to(
                    device), torch.Tensor(next_done).to(device)

                with torch.no_grad():
                    rm_ensamble_rewards = torch.stack(
                        [rm.forward(obs[step], actions[step]) for rm in reward_models])
                    rm_ensamble_rewards = torch.mean(
                        rm_ensamble_rewards, dim=0)

                    rm_rewards[step] = rm_ensamble_rewards.to(device).view(-1)
                buffer_1.store(obs[step], actions[step], gt_rewards[step],
                               next_obs, next_done, values[step], logprobs[step])

            for i in range(args.num_preference_pairs):
                if args.random_sample:
                    for seg_1 in buffer_1.sample(args.num_segments):
                        observations_1 = seg_1[0].reshape(-1,
                                                          seg_1[0].shape[1])
                        actions_1 = seg_1[1].reshape(-1, seg_1[1].shape[1])
                        rewards_1 = seg_1[2].reshape(-1)
                        log_probs_1 = seg_1[3].reshape(-1)
                        returns_1 = seg_1[4].reshape(-1)

                    for seg_2 in buffer_1.sample(args.num_segments):
                        observations_2 = seg_2[0].reshape(-1,
                                                          seg_2[0].shape[1])
                        actions_2 = seg_2[1].reshape(-1, seg_2[1].shape[1])
                        rewards_2 = seg_2[2].reshape(-1)
                        log_probs_2 = seg_2[3].reshape(-1)
                        returns_2 = seg_2[4].reshape(-1)
                else:
                    idx, segment_inds = buffer_1.reward_based_inds(
                        args.num_preference_pairs, args.num_segments)
                    for seg_1 in buffer_1.reward_based_sample(segment_inds[idx[0]]):
                        observations_1 = seg_1[0].reshape(-1,
                                                          seg_1[0].shape[1])
                        actions_1 = seg_1[1].reshape(-1, seg_1[1].shape[1])
                        rewards_1 = seg_1[2].reshape(-1)
                        log_probs_1 = seg_1[3].reshape(-1)
                        returns_1 = seg_1[4].reshape(-1)

                    for seg_2 in buffer_1.reward_based_sample(segment_inds[idx[-1]]):
                        observations_2 = seg_2[0].reshape(-1,
                                                          seg_2[0].shape[1])
                        actions_2 = seg_2[1].reshape(-1, seg_2[1].shape[1])
                        rewards_2 = seg_2[2].reshape(-1)
                        log_probs_2 = seg_2[3].reshape(-1)
                        returns_2 = seg_2[4].reshape(-1)

                    idx.pop(0)
                    idx.pop(-1)

                if args.human_feedback:
                    video_dir = f"pref_videos/pretraining/step_{step_pt}/pref_pair_{i+1}"
                    render_actions(
                        args.env_id, seg_1[1], "./" + video_dir, seg_nr=1, fps=30)
                    render_actions(
                        args.env_id, seg_2[1], "./" + video_dir, seg_nr=2, fps=30)
                    labels = buffer_1.compare_and_label_hf(
                        delete_videos=args.delete_videos, video_loading_dir=video_dir)
                else:
                    labels = buffer_1.compare_and_label(seg_1, seg_2)

                obs_action_1 = buffer_1.get_obs_action(seg_1)
                obs_action_2 = buffer_1.get_obs_action(seg_2)

                preference_pairs.append(
                    (obs_action_1[0], obs_action_1[1], obs_action_2[0], obs_action_2[1], labels[0], labels[1]))

            cross_ent_loss = rm_loss(preference_pairs=preference_pairs)

            for rm_optimizer in optimizers:
                rm_optimizer.zero_grad()

            cross_ent_loss.backward()

            for rm in reward_models:
                nn.utils.clip_grad_norm_(rm.parameters(), args.max_grad_norm)

            for rm_optimizer in optimizers:
                rm_optimizer.step()
            buffer_1.reset()

    total_feedback_calls = num_train_preference_pairs // args.num_preference_pairs
    feedback_interval = max(1, args.num_iterations // total_feedback_calls)
    executed_feedback_count = 0
    executed_validation_count = 0

    # matplotlip setup
    reward_array = np.zeros(args.total_timesteps)

    # start of the main training loop for the reward model update and PPO.
    for iteration in tqdm(range(1, args.num_iterations + 1), desc=f"Training over {args.num_iterations} iterations..."):
        # Annealing the rate if instructed to do so.
        if args.anneal_lr:
            frac = 1.0 - (iteration - 1.0) / args.num_iterations
            lrnow = frac * args.learning_rate
            optimizer.param_groups[0]["lr"] = lrnow

        if args.anneal_rm_lr:
            for rm_opt in optimizers:
                frac = 1.0 - (iteration - 1.0) / args.num_iterations
                lrnow_rm = frac * args.rm_learning_rate
                rm_opt.param_groups[0]["lr"] = lrnow_rm

        for step in range(0, args.num_steps):
            global_step += args.num_envs
            obs[step] = next_obs
            dones[step] = next_done

            # ALGO LOGIC: action logic
            with torch.no_grad():
                action, logprob, _, value = agent.get_action_and_value(
                    next_obs)
                values[step] = value.flatten()
            actions[step] = action
            logprobs[step] = logprob

            # TRY NOT TO MODIFY: execute the game and log data.
            next_obs, gt_reward, terminations, truncations, infos = envs.step(
                action.cpu().numpy())

            next_done = np.logical_or(terminations, truncations)
            gt_rewards[step] = torch.tensor(gt_reward).to(device).view(-1)
            next_obs, next_done = torch.Tensor(next_obs).to(
                device), torch.Tensor(next_done).to(device)

            with torch.no_grad():
                rm_ensamble_rewards = torch.stack(
                    [rm.forward(obs[step], actions[step]) for rm in reward_models])
                rm_ensamble_rewards = torch.mean(rm_ensamble_rewards, dim=0)

                rm_rewards[step] = rm_ensamble_rewards.to(device).view(-1)

            buffer_1.store(obs[step], actions[step], gt_rewards[step],
                           next_obs, next_done, values[step], logprobs[step])

            if "final_info" in infos:
                for info in infos["final_info"]:
                    if info and "episode" in info:
                        extracted_reward = info["episode"]["r"]
                        if isinstance(extracted_reward, list):
                            reward_array[global_step] = extracted_reward[0]
                        else:
                            reward_array[global_step] = extracted_reward.item()
                        writer.add_scalar(
                            "charts/episodic_return", info["episode"]["r"], global_step)
                        writer.add_scalar(
                            "charts/episodic_length", info["episode"]["l"], global_step)
                        
        rm_rewards_flat = rm_rewards.view(-1)
        gt_rewards_flat = gt_rewards.view(-1)

        #metrics
        pearson = compute_pearson_correlation(rm_rewards_flat, gt_rewards_flat)
        writer.add_scalar('Error/Pearson', pearson, global_step)

        mse = compute_mse(rm_rewards_flat, gt_rewards_flat)
        writer.add_scalar('Correlation/MSE', mse, global_step)

        # bootstrap value if not done
        with torch.no_grad():
            next_value = agent.get_value(next_obs).reshape(1, -1)
            advantages = torch.zeros_like(rm_rewards).to(device)
            lastgaelam = 0
            for t in reversed(range(args.num_steps)):
                if t == args.num_steps - 1:
                    nextnonterminal = 1.0 - next_done
                    nextvalues = next_value
                else:
                    nextnonterminal = 1.0 - dones[t + 1]
                    nextvalues = values[t + 1]
                delta = rm_rewards[t] + args.gamma * \
                    nextvalues * nextnonterminal - values[t]
                advantages[t] = lastgaelam = delta + args.gamma * \
                    args.gae_lambda * nextnonterminal * lastgaelam
            returns = advantages + values

        # flatten the batch_1
        b_obs = obs.reshape((-1,) + envs.single_observation_space.shape)
        b_logprobs = logprobs.reshape(-1)
        b_actions = actions.reshape((-1,) + envs.single_action_space.shape)
        b_advantages = advantages.reshape(-1)
        b_returns = returns.reshape(-1)
        b_values = values.reshape(-1)

        # Optimizing the policy and value network
        b_inds = np.arange(args.batch_size)
        clipfracs = []
        for epoch in range(args.update_epochs):
            np.random.shuffle(b_inds)
            for start in range(0, args.batch_size, args.minibatch_size):
                end = start + args.minibatch_size
                mb_inds = b_inds[start:end]

                _, newlogprob, entropy, newvalue = agent.get_action_and_value(
                    b_obs[mb_inds], b_actions.long()[mb_inds]
                )

                logratio = newlogprob - b_logprobs[mb_inds]
                ratio = logratio.exp()

                with torch.no_grad():
                    # calculate approx_kl http://joschu.net/blog/kl-approx.html
                    old_approx_kl = (-logratio).mean()
                    approx_kl = ((ratio - 1) - logratio).mean()
                    clipfracs += [((ratio - 1.0).abs() >
                                   args.clip_coef).float().mean().item()]

                loss, pg_loss, v_loss, entropy_loss, logratio = loss_fn(
                    newlogprob=newlogprob,
                    oldlogprob=b_logprobs[mb_inds],
                    advantages=b_advantages[mb_inds],
                    newvalue=newvalue,
                    oldvalue=b_values[mb_inds],
                    returns=b_returns[mb_inds],
                    entropy=entropy
                )

                optimizer.zero_grad()
                loss.backward()
                nn.utils.clip_grad_norm_(
                    agent.parameters(), args.max_grad_norm)
                optimizer.step()

            if args.target_kl is not None and approx_kl > args.target_kl:
                break
        # Update reward model and sample preference pairs
        if executed_feedback_count < num_train_preference_pairs and iteration % feedback_interval == 0:
            for i in range(args.num_preference_pairs):
                if args.random_sample:
                    for seg_1 in buffer_1.sample(args.num_segments):
                        observations_1 = seg_1[0].reshape(-1,
                                                          seg_1[0].shape[1])
                        actions_1 = seg_1[1].reshape(-1, seg_1[1].shape[1])
                        rewards_1 = seg_1[2].reshape(-1)
                        log_probs_1 = seg_1[3].reshape(-1)
                        returns_1 = seg_1[4].reshape(-1)

                    for seg_2 in buffer_1.sample(args.num_segments):
                        observations_2 = seg_2[0].reshape(-1,
                                                          seg_2[0].shape[1])
                        actions_2 = seg_2[1].reshape(-1, seg_2[1].shape[1])
                        rewards_2 = seg_2[2].reshape(-1)
                        log_probs_2 = seg_2[3].reshape(-1)
                        returns_2 = seg_2[4].reshape(-1)
                else:
                    idx, segment_inds = buffer_1.reward_based_inds(
                        args.num_preference_pairs, args.num_segments)
                    for seg_1 in buffer_1.reward_based_sample(segment_inds[idx[0]]):
                        observations_1 = seg_1[0].reshape(-1,
                                                          seg_1[0].shape[1])
                        actions_1 = seg_1[1].reshape(-1, seg_1[1].shape[1])
                        rewards_1 = seg_1[2].reshape(-1)
                        log_probs_1 = seg_1[3].reshape(-1)
                        returns_1 = seg_1[4].reshape(-1)

                    for seg_2 in buffer_1.reward_based_sample(segment_inds[idx[-1]]):
                        observations_2 = seg_2[0].reshape(-1,
                                                          seg_2[0].shape[1])
                        actions_2 = seg_2[1].reshape(-1, seg_2[1].shape[1])
                        rewards_2 = seg_2[2].reshape(-1)
                        log_probs_2 = seg_2[3].reshape(-1)
                        returns_2 = seg_2[4].reshape(-1)

                    idx.pop(0)
                    idx.pop(-1)

                if args.human_feedback:
                    video_dir = f"pref_videos/feedback_round_{executed_feedback_count+1}/pref_pair_{i+1}"
                    render_actions(
                        args.env_id, seg_1[1], "./" + video_dir, seg_nr=1, fps=30)
                    render_actions(
                        args.env_id, seg_2[1], "./" + video_dir, seg_nr=2, fps=30)
                    labels = buffer_1.compare_and_label_hf(
                        delete_videos=args.delete_videos, video_loading_dir=video_dir)
                else:
                    labels = buffer_1.compare_and_label(seg_1, seg_2)

                obs_action_1 = buffer_1.get_obs_action(seg_1)
                obs_action_2 = buffer_1.get_obs_action(seg_2)

                preference_pairs.append(
                    (obs_action_1[0], obs_action_1[1], obs_action_2[0], obs_action_2[1], labels[0], labels[1]))

            executed_feedback_count += 1

            # validation loss every 2th feedback. See tensorboard graphs to identify possible overfitting.
            if executed_feedback_count % 2 == 0 and executed_validation_count < num_validation_pairs:
                for i in range(args.num_preference_pairs):
                    if args.random_sample:
                        for seg_1 in buffer_1.sample(args.num_segments):
                            observations_1 = seg_1[0].reshape(
                                -1, seg_1[0].shape[1])
                            actions_1 = seg_1[1].reshape(-1, seg_1[1].shape[1])
                            rewards_1 = seg_1[2].reshape(-1)
                            log_probs_1 = seg_1[3].reshape(-1)
                            returns_1 = seg_1[4].reshape(-1)

                        for seg_2 in buffer_1.sample(args.num_segments):
                            observations_2 = seg_2[0].reshape(
                                -1, seg_2[0].shape[1])
                            actions_2 = seg_2[1].reshape(-1, seg_2[1].shape[1])
                            rewards_2 = seg_2[2].reshape(-1)
                            log_probs_2 = seg_2[3].reshape(-1)
                            returns_2 = seg_2[4].reshape(-1)
                    else:
                        idx, segment_inds = buffer_1.reward_based_inds(
                            args.num_preference_pairs, args.num_segments)
                        for seg_1 in buffer_1.reward_based_sample(segment_inds[idx[0]]):
                            observations_1 = seg_1[0].reshape(
                                -1, seg_1[0].shape[1])
                            actions_1 = seg_1[1].reshape(-1, seg_1[1].shape[1])
                            rewards_1 = seg_1[2].reshape(-1)
                            log_probs_1 = seg_1[3].reshape(-1)
                            returns_1 = seg_1[4].reshape(-1)

                        for seg_2 in buffer_1.reward_based_sample(segment_inds[idx[-1]]):
                            observations_2 = seg_2[0].reshape(
                                -1, seg_2[0].shape[1])
                            actions_2 = seg_2[1].reshape(-1, seg_2[1].shape[1])
                            rewards_2 = seg_2[2].reshape(-1)
                            log_probs_2 = seg_2[3].reshape(-1)
                            returns_2 = seg_2[4].reshape(-1)

                        idx.pop(0)
                        idx.pop(-1)

                    labels = buffer_1.compare_and_label(seg_1, seg_2)
                    obs_action_1 = buffer_1.get_obs_action(seg_1)
                    obs_action_2 = buffer_1.get_obs_action(seg_2)
                    validation_pairs.append(
                        (obs_action_1[0], obs_action_1[1], obs_action_2[0], obs_action_2[1], labels[0], labels[1]))

                executed_validation_count += 1
                with torch.no_grad():
                    validation_loss = rm_loss(
                        preference_pairs=validation_pairs)
                    scaled_val__loss = validation_loss.item() / len(validation_pairs)

                writer.add_scalar("losses/validation_loss",
                                  scaled_val__loss, global_step)

            cross_ent_loss = rm_loss(preference_pairs=preference_pairs)
            scaled_train_loss = cross_ent_loss.item() / len(preference_pairs)
            writer.add_scalar("losses/train_loss",
                              scaled_train_loss, global_step)

            for rm_optimizer in optimizers:
                rm_optimizer.zero_grad()

            cross_ent_loss.backward()

            for rm in reward_models:
                nn.utils.clip_grad_norm_(rm.parameters(), args.max_grad_norm)

            nn.utils.clip_grad_norm_(
                rm.parameters(), args.max_grad_norm)
            optimizer.step()
            rm_optimizer.step()

            for rm_optimizer in optimizers:
                rm_optimizer.step()

        buffer_1.reset()
