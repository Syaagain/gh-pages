class PPOLoss(nn.Module):
    def __init__(self, clip_coef: float = 0.2, clip_vloss: bool = True, ent_coef: float = 0.01, vf_coef: float = 0.5, norm_adv: bool = True):
        """This function implements the loss copmutation for Proximal Policy optimization (PPO). Implemented as nn.Module.

        Parameters
        ----------
        clip_coef : float, optional
            Clipping coefficient for the policy loss, prevents the probability ratio from becoming too large and createing updateds that would be too large, by default 0.2
        clip_vloss : bool, optional
            Wether or not to apply clipping to the value function loss, by default True
        ent_coef : float, optional
            Coefficient for the entropy loss, higher values: more exploration by penalizing policies that are to deterministic, by default 0.01
        vf_coef : float, optional
            Coefficient for the value function loss. Weights the value function loss within the total loss computation by a factor, by default 0.5
        norm_adv : bool, optional
            Wether to normalize the advantages before computing the policy loss, by default True
        """
        super(PPOLoss, self).__init__()
        self.clip_coef = clip_coef
        self.clip_vloss = clip_vloss
        self.ent_coef = ent_coef
        self.vf_coef = vf_coef
        self.norm_adv = norm_adv

    def forward(self, newlogprob: torch.Tensor, oldlogprob: torch.Tensor, advantages: torch.Tensor, newvalue: torch.Tensor, oldvalue: torch.Tensor, returns: torch.Tensor, entropy: torch.Tensor) -> Tuple[torch.Tensor, torch.Tensor, torch.Tensor, torch.Tensor, torch.Tensor]:
        """Forward pass of the PPOLoss function module, this handles the loss computation. See "Retruns" section for a detailed view on the return values
        and how to use them.

        Parameters
        ----------
        newlogprob : torch.Tensor
            Log probabilities of actions taken under the current policy. (updated policy)
        oldlogprob : torch.Tensor
            Log probabilities of actions taken under the previous policy. (old policy )
        advantages : torch.Tensor
            Generalized Advantage Estimation (GAE) values, see compute_gae function in utils/value_estimation.py for more details.
        newvalue : torch.Tensor
            Predicted state values under the current policy.
        oldvalue : torch.Tensor
            Predicted state values under the old policy.
        returns : torch.Tensor
            Target state values, usually computed as the sum of rewards plus discounted future returns.
        entropy : torch.Tensor
            Entropy of the current policy, measure of randomness in the action distribution.

        Returns
        -------
        Tuple[torch.Tensor, torch.Tensor,
            torch.Tensor, torch.Tensor, torch.Tensor]
            total loss (policy_loss, value_loss and entropy_loss),
            policy gradient loss (how well the new poliy improves action selection, of course based on the advantages),
            value function loss (how well the value estimates match the returns),
            entropy loss,
            logratio (log probs between the new and old polices -> measure change in the policies)
        """
        if self.norm_adv:
            advantages = (advantages - advantages.mean()) / \
                (advantages.std() + 1e-8)

        logratio = newlogprob - oldlogprob
        ratio = logratio.exp()
        pg_loss1 = -advantages * ratio
        pg_loss2 = -advantages * \
            torch.clamp(ratio, 1 - self.clip_coef, 1 + self.clip_coef)
        pg_loss = torch.max(pg_loss1, pg_loss2).mean()

        newvalue = newvalue.view(-1)
        if self.clip_vloss:
            v_loss_unclipped = (newvalue - returns) ** 2
            v_clipped = oldvalue + \
                torch.clamp(newvalue-oldvalue, -self.clip_coef, self.clip_coef)
            v_loss_clipped = (v_clipped - returns) ** 2
            v_loss_max = torch.max(v_loss_unclipped, v_loss_clipped)
            v_loss = v_loss_max.mean()
        else:
            v_loss = 0.5 * ((newvalue - returns) ** 2).mean()

        entropy_loss = entropy.mean()

        total_loss = pg_loss - self.ent_coef * entropy_loss + v_loss * self.vf_coef

        return total_loss, pg_loss, v_loss, entropy_loss, logratio

class PreferenceLoss(nn.Module):
    def __init__(self, reward_models: Union[nn.Module, List[nn.Module]], device: torch.device, entropy_bonus: float = 0.01, normalize: bool = True, batch_size: int = 64):
        """Calculation for the PreferenceLoss. Calculate the reward for the Reward Model based on a set of preference pairs. The implementation features the capability of useing a reward model
        ensamble, to migate the influence of bad predictions, by averaging the final reward prediction of state/actions over the number of reward models.
        This implementation also features batched input, to efficiently predict the rewards for observation action pairs not only over multiple frames, but also batch up to batch_size preference pairs
        along  NOTE: Ensure the reward model and loss module are on the same device.

        Parameters
        ----------
        reward_model : nn.Module, List[nn.Module]
            Reward model (ensamble if more then one) that should be used for the reward calculation.
        device : torch.device
            Specify on which device the calculation should happen. If no value is given it will always run on a gpu if one is available.
        entropy_bonus: float, optional
            This bonus helps with incentivizing the increased exploration needed, when dealing with a changing reward function, by defualt 0.01
        normalize: bool, optional
            Wether or not the reward calculations should be normalized to have a standard deviation of 1, by default True
        batch_size:int, optional
            Batch size, that specifies the max value of how many trajectory pairs can be packed into one batch.

        Examples
        --------

        reward_model1 = RewardModel(envs=envs).to(device)
        reward_model2 = RewardModel(envs=envs).to(device)
        pref_loss = PreferenceLoss(
            reward_models=[reward_model1, reward_model2], device=device).to(device)

        """

        super(PreferenceLoss, self).__init__()
        if not isinstance(reward_models, list):
            reward_models = [reward_models]
        self.reward_models = reward_models
        self.entropy_bonus = entropy_bonus
        self.normalize = normalize
        if device is None:
            self.device = torch.device(
                "cuda" if torch.cuda.is_available() else "cpu")
        else:
            self.device = device
        self.batch_size = batch_size

    def bradley_terry_model(self, trajectory1: list[torch.Tensor, torch.Tensor], trajectory2: list[torch.Tensor, torch.Tensor], reward_models: Union[nn.Module, List[nn.Module]]) -> Tuple[torch.Tensor, torch.Tensor]:
        """This function is the core of the preference caluclation, it returns the softmax of which trajectory segment is preffered over the other one.
        The calculation is processed in a batched like manner, to speed up reward model training. This means you get prediction pairs in a stacked manner.
        This also follows the approach of fitting the reward prediction in "Deep Reinforcement Learning from Human Preferences" by Christiano et. al.: https://arxiv.org/pdf/1706.03741.

        Parameters
        ----------
        trajectory1 : list[torch.Tensor, torch.Tensor]
            Batch of first trajectories.
        trajectory2 : list[torch.Tensor, torch.Tensor]
            Batch of second trajectories.
        reward_models : Union[nn.Module, List[nn.Module]]
            All reward models that should be used for the calculation/training.

        Returns
        -------
        Tuple[torch.Tensor, torch.Tensor]
            Batched tensor each, with the softmax/prefernce predicionts of trajectory 1 over 2 and vice versa. Contains paired values in respect to the batch size.
        """
        all_rewards_1 = []
        all_rewards_2 = []

        for reward_model in reward_models:
            rewards_1 = torch.stack([reward_model(obs, act) for obs, act in zip(
                trajectory1[0], trajectory1[1])]).to(self.device)
            rewards_2 = torch.stack([reward_model(obs, act) for obs, act in zip(
                trajectory2[0], trajectory2[1])]).to(self.device)

            if self.normalize:
                std1 = rewards_1.std(dim=0, keepdim=True) + 1e-8
                mean1 = rewards_1.mean(dim=0, keepdim=True)
                std2 = rewards_2.std(dim=0, keepdim=True) + 1e-8
                mean2 = rewards_2.mean(dim=0, keepdim=True)
                normalized_rewards_1 = (rewards_1 - mean1) / std1
                normalized_rewards_2 = (rewards_2 - mean2) / std2

                all_rewards_1.append(normalized_rewards_1)
                all_rewards_2.append(normalized_rewards_2)
            else:
                all_rewards_1.append(rewards_1)
                all_rewards_2.append(rewards_2)

        all_rewards_1 = torch.stack(all_rewards_1, dim=1)
        all_rewards_2 = torch.stack(all_rewards_2, dim=1)

        average_rewards_1 = torch.mean(all_rewards_1, dim=1)
        average_rewards_2 = torch.mean(all_rewards_2, dim=1)

        total_reward_1 = torch.sum(average_rewards_1, dim=1)
        total_reward_2 = torch.sum(average_rewards_2, dim=1)

        total_rewards = torch.stack(
            [total_reward_1, total_reward_2], dim=0)

        prediction_r_1_2, prediction_r_2_1 = torch.softmax(
            total_rewards, dim=0)

        return prediction_r_1_2, prediction_r_2_1

    def forward(self, preference_pairs: list, entropy_bonus: bool = True, random_shuffle: bool = True) -> torch.Tensor:
        """Function to calculate the preference reward over a Dataset of trajectories.
        Follows the implementation of 2.2.3 of "Deep Reinforcement Learning from Human Preferences"
        by Christiano et. al.: https://arxiv.org/pdf/1706.03741.

        Parameters
        ----------
        preference_pairs : list
            List containin tensors for trajectories and labels, size of list and shape of tensors: [(obs_action_1[0], obs_action_1[1], obs_action_2[0], obs_action_2[1], labels[0], labels[1]), ...]
        entropy_bonus: bool, optional
            Wether or not a entropy bonus should be applied. Set value during initalisation of this class, by default True
        random_shuffle: bool, optional
            Wether or not the list of preference pairs should be suffled each time, to generate varying batches each time, defaults to True
        Returns
        -------
        torch.Tensor
            Returns the cross entropy loss for a batch of trajectory predictions and their assigned human or synthetic labels.
        """
        cross_entropy_loss = 0
        if random_shuffle:
            random.shuffle(preference_pairs)

        batches = [preference_pairs[i:i + self.batch_size]
                   for i in range(0, len(preference_pairs), self.batch_size)]

        for batch in batches:
            obs_1_batch = torch.stack([pair[0] for pair in batch]).to(
                self.device)
            actions_1_batch = torch.stack([pair[1] for pair in batch]).to(
                self.device)
            obs_2_batch = torch.stack([pair[2] for pair in batch]).to(
                self.device)
            actions_2_batch = torch.stack([pair[3] for pair in batch]).to(
                self.device)
            mu1_batch = torch.stack(
                [pair[4] for pair in batch])
            mu2_batch = torch.stack(
                [pair[5] for pair in batch])

            trajectory1_batch = [obs_1_batch, actions_1_batch]
            trajectory2_batch = [obs_2_batch, actions_2_batch]

            prediction_r_1_2, prediction_r_2_1 = self.bradley_terry_model(
                trajectory1=trajectory1_batch,
                trajectory2=trajectory2_batch,
                reward_models=self.reward_models
            )

            mu1_batch = mu1_batch.view(-1, 1)
            mu2_batch = mu2_batch.view(-1, 1)

            preference_t_batch = (
                mu1_batch * torch.log(prediction_r_1_2) +
                mu2_batch * torch.log(prediction_r_2_1)
            )
            cross_entropy_loss += torch.sum(preference_t_batch)

        if entropy_bonus:
            cross_entropy_loss += entropy_bonus
        return -cross_entropy_loss
